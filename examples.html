// 基础使用示例 const agent = new LitoAgent(); // 处理用户请求 const result = await agent.processRequest( "当Lito被忽视时，会委屈地碎碎念，头微微低下", { persona: 'friendly', evolution: { daysRaised: 50 } } ); // 检查结果 if (result.success) { console.log('VAT:', result.data.vat); console.log('马达参数:', result.data.motorParameters); console.log('安全等级:', result.data.safetyLevel); // 导出为 JSON const exportResult = agent.lereExporter.exportToJSON(result.data.behavior); console.log('导出文件:', exportResult.filename); }
// 高级使用示例 - 自定义处理 const agent = new LitoAgent(); // 1. 先进行歧义检查 const ambiguityCheck = agent.boundaryHandler.handleAmbiguity(userInput); if (ambiguityCheck.hasAmbiguity) { console.log('需要澄清:', ambiguityCheck.ambiguities); } // 2. 检查风险操作 const riskCheck = agent.boundaryHandler.checkRiskOperation(userInput); if (riskCheck.hasRisk) { console.log('风险操作:', riskCheck.risks); } // 3. 创建备份 await agent.boundaryHandler.backupManager.createBackup( currentData, 'behavior_modification' ); // 4. 执行处理 const result = await agent.processRequest(userInput, options); // 5. 评估不确定性 const uncertainty = agent.boundaryHandler.assessUncertainty(result); if (uncertainty.requiresWarning) { console.log('警告:', uncertainty.warningMessage); }
